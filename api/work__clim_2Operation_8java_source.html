151  setValue(intesisBox,aceco,1,1);

163  }

165  }
166  // Regulate Fan depending on in/out humidity
167 
168  // humPoint achieved within humDelta
169  if (Math.abs(humPoint - humIn.value) <= humDelta) {
170  setValue(pwm,vOut,0,0);
171  setValue(intesisBox,aceco,1,1);

173 
174  // humIn < humPoint; between n & (n+1) humDelta

176  if ((humPoint - humIn.value) <= humDelta*(n+1) && (humPoint - humIn.value) > humDelta*n) {
177  logger.finer("Operation:operationAuto50> humOut.value = " +humOut.value + "humIn.value = " +humIn.value);
178  if (humOut.value > humIn.value)
179  setValue(pwm,vOut,0,Math.min(n,pwm.vMax));

181  logger.finer("Operation:operationAuto5> humOut.value = " +humOut.value + "humIn.value = " +humIn.value);
182  setValue(pwm,vOut,0,0);
183  setValue(intesisBox,aceco,1,1);
184  setValue(intesisBox,acfan,1,0);
185  }
186  }
187  }
188  // humIn > humPoint; between n & (n+1) humDelta
189  for (int n = 1; n < 10; n++) {
190  if ((humIn.value - humPoint) <= humDelta*(n+1) && (humIn.value - humPoint) > humDelta*n) {
191  logger.finer("Operation:operationAuto6> humOut.value = " +humOut.value + "humIn.value = " +humIn.value);
