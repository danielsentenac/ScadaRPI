214  String.format("%32s",Integer.toBinaryString(i2c_buffer)).replaceAll(" ", "0"));
215

224  if ( e.value == 1 ) // Open Valve

226  else if ( e.value == 2 ) // Close Valve
227  i2c_buffer = clearBit(i2c_buffer,VENTSOFT_CLOSE_CMD_BIT);
228  }
229  else if ( e.name.contains("VPCMD") ) { // Valve VP Open/Close command
230  logger.finer("Controllino_2:executeCommand> set bit VP =" + e.value);
231  if ( e.value == 1 ) // Open Valve
232  i2c_buffer = setBit(i2c_buffer,VP_OPEN_CMD_BIT);
233  else if ( e.value == 2 ) // Close Valve
234  i2c_buffer = clearBit(i2c_buffer,VP_CLOSE_CMD_BIT);
235  }
236  else if ( e.name.contains("VSPAREONOFF") ) { // VSPARE On/Off command
237  logger.finer("Controllino_2:executeCommand> set bit VSPARE =" + e.value);
238  if ( e.value == 1 ) // Switch VSPARE On
239  i2c_buffer = setBit(i2c_buffer,VSPARE_ON_CMD_BIT);
240  else if ( e.value == 2 ) // Switch VSPARE Off

245  if ( e.value == 1 ) // Switch BYPASS On
246  i2c_buffer = setBit(i2c_buffer,BYPASS_ON_CMD_BIT);

251  logger.fine("Controllino_2:executeCommand> send i2c_buffer=" +
252  String.format("%32s",Integer.toBinaryString(i2c_buffer)).replaceAll(" ", "0"));
253  // Write the (new) i2c_buffer on the bus
