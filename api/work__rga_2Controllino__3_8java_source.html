153  public void executeCommand( DataElement e ) {
154 
155  try {

157  byte[] serDataR = null;

171  String.format("%32s",Integer.toBinaryString(i2c_buffer)).replaceAll(" ", "0"));
172 
173  if ( e.name.contains("FANSPEED") ) { // Fan speed command
174  if ( e.setvalue == 1 ) { // Normal speed
175  logger.finer("Controllino_3:executeCommand> SET NORMAL SPEED");
176  i2c_buffer = clearBit(i2c_buffer,LOW_NOISE_OPEN_CMD_BIT);
177  }
178  else if ( e.setvalue == 2 ) { // Low noise
179  i2c_buffer = clearBit(i2c_buffer,NORMAL_SPEED_CLOSE_CMD_BIT);

183  if ( e.value == 1 ) // Start Fan

187  }
188  else if ( e.name.contains("RESET3") ) { // RESET Controllino 3 command
189  if ( e.value == 1 ) // Rest Controllino 3
190  i2c_buffer = setBit(i2c_buffer,ARD_RESET_BIT);
191  }
192 
193  logger.fine("Controllino_3:executeCommand> send i2c_buffer=" +
194  String.format("%32s",Integer.toBinaryString(i2c_buffer)).replaceAll(" ", "0"));
195  // Write the (new) i2c_buffer on the bus
196  ByteBuffer buffer = ByteBuffer.allocate(4);
197  buffer.putInt(i2c_buffer);
