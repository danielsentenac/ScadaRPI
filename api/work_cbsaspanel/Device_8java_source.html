95  DataElement dataElement = e.getValue();
96  if ( dataElement.type == DataType.TRIGGER ) { // Trigger command

98  dataElement.setvalue = holdingRegisters.getInt16At(dataElement.mbRegisterOffset);
99  if ( dataElement.setvalue != 0 ) {// Command has been triggered by Modbus Master --> add it
100  logger.fine("Device:addModbusCommand>TRIGGER: Pushing Modbus command for "
101  + dataElement.name + " with value " + dataElement.setvalue);
102  commandSetQueue.add(dataElement);
103  }
104  }
105  catch ( IllegalDataAddressException ex) {
106  logger.log(Level.SEVERE, "Device:addModbusCommands> " + ex.getMessage());
107  }
108  }
109  else if ( dataElement.type == DataType.READ_AND_WRITE_VALUE ||
110  dataElement.type == DataType.READ_AND_WRITE_STATUS ) { // Writable data from modbus
111  try {
112  // Set new reference value from holdingRegisters
113  switch (dataElement.mbRegisterType) {
114  case INT8:
115  dataElement.setvalue = holdingRegisters.getInt8At(dataElement.mbRegisterOffset);
116  break;
117  case INT16:
118  dataElement.setvalue = holdingRegisters.getInt16At(dataElement.mbRegisterOffset);
119  break;
120  case INT32:
121  dataElement.setvalue = holdingRegisters.getInt32At(dataElement.mbRegisterOffset);
122  break;
123  case FLOAT32:
124  dataElement.setvalue = holdingRegisters.getFloat32At(dataElement.mbRegisterOffset);
125  break;
126  default: logger.log(Level.SEVERE, "Device:addModbusCommand> Invalid data element type");
127  break;
128  }
129  if (dataElement.value != dataElement.setvalue) {
130  logger.fine("Device:addModbusCommand>READ_AND_WRITE_VALUE: Pushing Modbus command for "
131  + dataElement.name + " with value " + dataElement.setvalue );
132  commandSetQueue.add(dataElement);
133  }
134  }
