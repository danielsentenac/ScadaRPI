188  public void executeCommand( DataElement e ) {
189 
190  try {
191  // Read the (old) i2c buffer from the bus
192  byte[] serDataR = i2c.Read();
193  int i2c_buffer = ByteBuffer.wrap(serDataR).getInt();
194  logger.finer("Controllino:executeCommand> reading i2c_buffer=" +
195  String.format("%32s",Integer.toBinaryString(i2c_buffer)).replaceAll(" ", "0"));
196 
197  if ( e.name.contains("V21CMD") ) { // Valve V21 Open/Close command
198  if ( e.value == 1 ) // Open Valve
199  i2c_buffer = setBit(i2c_buffer,V21_OPEN_CMD_BIT);
200  else if ( e.value == 2 ) // Close Valve
201  i2c_buffer = clearBit(i2c_buffer,V21_CLOSE_CMD_BIT);
202  }
203  else if ( e.name.contains("V22CMD") ) { // Valve V22 Open/Close command
204  logger.finer("Controllino:executeCommand> set bit V22 =" + e.value);
205  if ( e.value == 1 ) // Open Valve
206  i2c_buffer = setBit(i2c_buffer,V22_OPEN_CMD_BIT);
207  else if ( e.value == 2 ) // Close Valve
208  i2c_buffer = clearBit(i2c_buffer,V22_CLOSE_CMD_BIT);
209  }
210  else if ( e.name.contains("V31CMD") ) { // Valve V31 Open/Close command
211  logger.finer("Controllino:executeCommand> set bit V31 =" + e.value);
212  if ( e.value == 1 ) // Open Valve
213  i2c_buffer = setBit(i2c_buffer,V31_OPEN_CMD_BIT);
214  else if ( e.value == 2 ) // Close Valve
215  i2c_buffer = clearBit(i2c_buffer,V31_CLOSE_CMD_BIT);
216  }
217  else if ( e.name.contains("P22ONOFF") ) { // Scroll P22 On/Off command
218  logger.finer("Controllino:executeCommand> set bit P22 =" + e.value);
219  if ( e.value == 1 ) // Switch P22 On
220  i2c_buffer = setBit(i2c_buffer,P22_ON_CMD_BIT);
221  else if ( e.value == 2 ) // Switch P22 Off
222  i2c_buffer = clearBit(i2c_buffer,P22_OFF_CMD_BIT);
223  }
224  else if ( e.name.contains("P3132ONOFF") ) { // Titanium P31_32 On/Off command
225  logger.finer("Controllino:executeCommand> set bit P31_32 =" + e.value);
226  if ( e.value == 1 ) // Switch P31_32 On
227  i2c_buffer = setBit(i2c_buffer,P31_32_ON_CMD_BIT);
228  else if ( e.value == 2 ) // Switch P31_32 Off
229  i2c_buffer = clearBit(i2c_buffer,P31_32_OFF_CMD_BIT);
230  }
231 
232  logger.finer("Controllino:executeCommand> send i2c_buffer=" +
233  String.format("%32s",Integer.toBinaryString(i2c_buffer)).replaceAll(" ", "0"));
234  // Write the (new) i2c_buffer on the bus
235  ByteBuffer buffer = ByteBuffer.allocate(4);
236  buffer.putInt(i2c_buffer);
237  byte[] msg = buffer.array();
238  i2c.Write(msg);

246  setErrorComStatus();

248  }
