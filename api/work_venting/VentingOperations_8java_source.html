71  setOperationStep(0);
72 
73  for (int i = 0; i < steps.size(); i++) {
74  // global stop check at top of loop
75  if (stopRequested || Thread.currentThread().isInterrupted()) {
76  logger.info("VentingOperations: stop requested before step " + i + ", aborting sequence.");
77  break;
78  }
79 
80  Step step = steps.get(i);
81  boolean isLastStep = (i == steps.size() - 1);
82  setOperationStep(step.step);
83 
84  // 🔦 highlight active row
85  highlightRow(step.rowIndex);
86 
87  logger.info("VentingOperations: executing step " + step.step +
88  (isLastStep ? " (last step)" : ""));
89 
90  // 1) MKS flows
91  sendMksCommands(step);
92  if (stopRequested || Thread.currentThread().isInterrupted()) break;
93 
94  // 2) Valve states
95  sendValveCommand("M1_V1CMD", step.v1);
96  sendValveCommand("M1_VSOFTCMD", step.vsoft);
97  sendValveCommand("M1_VMAINCMD", step.vmain);

99  sendValveCommand("M1_VRPCMD", step.vrp);
100  if (stopRequested || Thread.currentThread().isInterrupted()) break;
101 
102  // 3) Pump P1
103  sendPumpCommand("M1_P1ONOFF", step.p1);
104  if (stopRequested || Thread.currentThread().isInterrupted()) break;
105 
106  // 4) Wait for pressure *only if there is a next step*
107  if (!isLastStep && step.g2Target > 0) {
108  boolean stopped = waitForPressure(step.g2Target);
109  if (stopped) {
110  logger.info("VentingOperations: stopped while waiting for pressure.");
111  break;
112  }
113  }
114  }
115 
116  setOperationStep(0);
117  logger.info("VentingOperations: sequence finished.");
