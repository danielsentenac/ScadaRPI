133  public void configureRegistersFromAllDevice(ModbusHoldingRegisters holdingRegisters) {
134 
135  logger.finer("ModbusSlaveThread:configureRegistersFromAllDevice> Configure Modbus Registers for all devices");
136 
137  for (Map.Entry<String, Device> d : deviceManager.deviceList.entrySet()) {
138  for (Map.Entry<String, DataElement> e : d.getValue().dataList.entrySet()) {
139  DataElement dataElement = e.getValue();
140  try {
141  switch (dataElement.mbRegisterType) {
142  case INT8: dataElement.mbRegisterLength = 1; // 8 bit Byte
143  holdingRegisters.set(dataElement.mbRegisterOffset, 0);
144  break;
145  case INT16: dataElement.mbRegisterLength = 1; // 16 bit Short
146  holdingRegisters.set(dataElement.mbRegisterOffset, 0);
147  break;
148  case INT32: dataElement.mbRegisterLength = 2; // 32 bit Integer
149  holdingRegisters.set(dataElement.mbRegisterOffset, 0);
150  holdingRegisters.set(dataElement.mbRegisterOffset+1, 0);
151  break;
152  case FLOAT32: dataElement.mbRegisterLength = 2; // 32 bit Float
153  holdingRegisters.set(dataElement.mbRegisterOffset, 0);
154  holdingRegisters.set(dataElement.mbRegisterOffset+1, 0);
155  break;
156  default: logger.log(Level.SEVERE, "ModbusSlaveThread:configureRegistersFromAllDevice> " +
157  dataElement.name + ":Invalid data element type:");
158  break;
159  }
160  }
161  catch ( IllegalDataAddressException ex) {
162  logger.log(Level.SEVERE, "ModbusSlaveThread:configureRegistersFromAllDevice> " +
163  d.getKey() + "," + dataElement.name + ":" + ex.getMessage() + ":" + ex.getDataAddress());
164  }
165  catch ( IllegalDataValueException ex) {
166  logger.log(Level.SEVERE, "ModbusSlaveThread:configureRegistersFromAllDevice> " +
167  d.getKey() + "," + dataElement.name + ":"+ ex.getMessage());
168  }
169  }
170  }
171  }

222  logger.fine("ModbusSlaveThread:writeHoldingRegister> Writing register at offset " + offset + " value=" + value);
223  for (ModbusEventListener l : modbusEventListenerList) {
224  l.onWriteToSingleHoldingRegister(offset, value);

227  Device device = d.getValue();

229  DataElement dataElement = device.getDataElement(offset);
230  logger.finer("ModbusSlaveThread:writeHoldingRegister> found dataElement:" + dataElement);
231  if (dataElement != null &&
232  ( dataElement.type == DataType.TRIGGER || // Command triggers registers
233  dataElement.type == DataType.READ_AND_WRITE_VALUE ||

235  logger.finer("ModbusSlaveThread:writeHoldingRegister> Acquire lock..");
236  device.modbusmutex.lock();
