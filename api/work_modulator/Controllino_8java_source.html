173  public void executeCommand( DataElement e ) {
174 
175  try {
176  // Read the (old) i2c buffer from the bus
177  byte[] serDataR = i2c.Read28bytesPlusCRC32();
178  if (serDataR == null) {
179  logger.log(Level.WARNING,"Controllino_3:executeCommand> NO COMMAND SENT! (data=null)");
180  return;
181  }
182  int i2c_buffer = ByteBuffer.wrap(Arrays.copyOfRange(serDataR, 20, 24)).getInt();
183  byte[] value = new byte[]{0,0};
184 
185  logger.finer("Controllino:executeCommand> reading i2c_buffer=" +
186  String.format("%32s",Integer.toBinaryString(i2c_buffer)).replaceAll(" ", "0"));
187 
188  if ( e.name.contains("MAINONOFF") ) { // MAIN ON/OFF command
189  if ( e.value == 1 ) // Switch ON
190  i2c_buffer = setBit(i2c_buffer,MAIN_ON_CMD_BIT);
191  else if ( e.value == 2 ) // Switch OFF
192  i2c_buffer = clearBit(i2c_buffer,MAIN_OFF_CMD_BIT);
193  }
194  else if ( e.name.contains("NEGATIVEVOLT") ) { // Set value NEGATIVE VOLTAGE command
195  short tmp = (short) e.setvalue;
196  value[0] = (byte)(( tmp >> 8) & 0xFF);
197  value[1] = (byte)(tmp & 0xFF);
198  // Set the trigger bit
199  i2c_buffer = setBit(i2c_buffer,NEGATIVE_VOLT_TRIG_BIT);
200  }
201  else if ( e.name.contains("POSITIVEVOLT") ) { // Set value POSITIVE VOLTAGE command
202  short tmp = (short) e.setvalue;
203  value[0] = (byte)(( tmp >> 8) & 0xFF);
204  value[1] = (byte)(tmp & 0xFF);
205  // Set the trigger bit
206  i2c_buffer = setBit(i2c_buffer,POSITIVE_VOLT_TRIG_BIT);
207  }
208  else if ( e.name.contains("FREQUENCY") ) { // Set value FREQUENCY command
209  short tmp = (short) e.setvalue;
210  value[0] = (byte)(( tmp >> 8) & 0xFF);
211  value[1] = (byte)(tmp & 0xFF);
212  // Set the trigger bit
213  i2c_buffer = setBit(i2c_buffer,FREQUENCY_TRIG_BIT);
214  }
215  else if ( e.name.contains("DUTYCYCLE") ) { // Set value DUTY CYCLE command
216  short tmp = (short) e.setvalue;
217  value[0] = (byte)(( tmp >> 8) & 0xFF);
218  value[1] = (byte)(tmp & 0xFF);
219  // Set the trigger bit
220  i2c_buffer = setBit(i2c_buffer,DUTYCYCLE_TRIG_BIT);
221  }
222  logger.finer("Controllino:executeCommand> send i2c_buffer=" +
223  String.format("%32s",Integer.toBinaryString(i2c_buffer)).replaceAll(" ", "0"));
224  // Write the i2c_buf on the bus
225  byte[] i2cbuf = new byte[]{0,0,0,0,0,0};
226  i2cbuf[0] = (byte)(( i2c_buffer >> 24) & 0xFF);
227  i2cbuf[1] = (byte)(( i2c_buffer >> 16) & 0xFF);
228  i2cbuf[2] = (byte)(( i2c_buffer >> 8) & 0xFF);
229  i2cbuf[3] = (byte)( (i2c_buffer >> 0 ) & 0xFF);
230  i2cbuf[4] = value[0];
231  i2cbuf[5] = value[1];
232  for (int i = 0 ; i < 6 ; i++)
233  logger.finer("SENDING BYTE BUFFER:" + String.format("%64s",Integer.toBinaryString(i2cbuf[i])).replaceAll(" ", "0"));
234  i2c.WritePlusCRC32(i2cbuf);
235  if (e.type == DataType.TRIGGER) {
236  Thread.sleep(2000); // Wait before resetting
237  e.value = 0;
238  holdingRegisters.setInt16At(e.mbRegisterOffset, 0);
239  }
240  }

242  logger.log(Level.SEVERE, "Controllino:executeCommand>" + ex.getMessage());
243  setErrorComStatus();

245  }
